#### html
```html

<!DOCTYPE html>
<html lang="ja">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>FashionablyLate</title>
    {{-- CSSは上から順に読み込まれる = 同じセレクタ/プロパティを複数回指定すると後に読み込まれたほうが勝つ --}}
    {{-- リセットCSS（ress） を最初に読み込み、ブラウザごとの余白や見出しの差をなくす--}}
    <link rel="stylesheet" href="https://unpkg.com/ress/dist/ress.min.css" />

    <link rel="stylesheet" href="{{ asset('css/variables.css') }}" />

    <link rel="stylesheet" href="{{ asset('css/register.css') }}" />
```

#### register.css
```css

/*---------------------------------
留意ポイント

1 レスポンシブ対応度を上げるため可変単位のみ使用する
2 ヘッダー内の要素配置はposition relativeではなくgridで安定性を上げる
3 rem,vh,vw,%の使い分けを意図的に行う
- rem     padding, margin, font-size
- vh,vw   幅や高さ（親要素の影響を受けない = 画面全体が親とみなせる場合）
- %       幅や高さ (親要素のサイズによって影響される場合)
---------------------------------*/

/*---------------------------------
ENTIRE PAGE SETTING
---------------------------------*/
/* 1rem = 10pxに設定しわかりやすくする */
html {
    font-size: 62.5%;
}
/* フォーム以外の全体背景色 */
body {
    background-color: #f1ece7;
}

/*---------------------------------
HEADER
---------------------------------*/
.header {
    height: 10vh;
    min-height: 8rem;
    display: grid;
    /* 横方向 = 列は3等分で全部1frずつ, repeat(3, 1fr) = 1fr 1fr 1fr */
    grid-template-columns: repeat(3, 1fr);
    grid-template-rows: 1fr;
    align-items: center;
    padding: 1.5rem 0;
    border: 1px solid #e0dfde;
    /* 各セル内のインライン要素を左右中央に配置 */
    text-align: center;
    background-color: white;
    color: var(--color-text);
}

.header__title {
    grid-column: 2;
    grid-row: 1;
    font-size: clamp(
        var(--fs-header_ttl-min),
        var(--fs-header_ttl-fluid),
        var(--fs-header_ttl-max)
    );
}

.header__link {
    /* ボタンらしい四角い見た目にする */
    padding: 1.5vh 2vw;
    /* ボタンの線を演出 */
    border: 1px solid #d9c6b5;
    /* ボタンの面を演出 */
    background-color: #f0f0f0;
    color: #d9c6b5;
    /* リンクの下線を消す */
    text-decoration: none;
    grid-column: 3;
    grid-row: 1;
    margin: auto;
}

/*---------------------------------
REGISTER FORM
---------------------------------*/

.register-form__header {
    text-align: center;
    padding-top: 2rem;
    font-size: clamp(
        var(--fs-main_ttl-min),
        var(--fs-main_ttl-fluid),
        var(--fs-main_ttl-max)
    );
}

.register-form__inner {
    display: flex;
    justify-content: center;
    margin: 4rem 0;
}

.register-form__form {
    /* 過剰な拡大縮小を防ぐ為max,min指定 */
    width: 90%; /* 画面幅を基準に90%を上限とする */
    max-width: 60rem; /* 600px以上には広がらない */
    min-width: 28rem; /* 280px以下には縮まない */
    border: solid 1px #c5b9b0;
    background-color: white;
    padding: 1.5rem 2rem;
}

.register-form__group {
    display: flex;
    flex-direction: column;
    margin: auto;
    padding: 2rem;
}

.register-form__label {
    font-size: clamp(
        var(--fs-form__label-min),
        var(--fs-form__label-fluid),
        var(--fs-form__label-max)
    );
}

.register-form__input {
    border: solid 1px lightgray;
    width: 100%;
    margin-top: 0.5rem;
    padding: 1rem;
}

.register-form_error {
    font-family: var(--ff-validation);
    color: var(--color-validation-text);
}

.register-form__button {
    display: var(--btn-display);
    border: var(--bd-btn);
    color: var(--clr-btn-txt);
    background: var(--clr-btn-bg);
    padding: var(--btn-padding-vertical) var(--btn-padding-horizontal);
    margin: var(--btn-margin);
}

.register-form__button:hover {
    filter: var(--clr-btn-hover);
}

/* キーボード使用時のフォーカス */
.register-form__input:focus,
.header__link:focus,
.register-form__button:focus {
    outline: 2px solid #6c63ff;
    outline-offset: 2px;
}
```
