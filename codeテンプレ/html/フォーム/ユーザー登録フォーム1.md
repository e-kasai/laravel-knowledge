### code

```html
<!DOCTYPE html>
<html lang="ja">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>FashionablyLate</title>
    <link rel="stylesheet" href="{{ asset('css/sanitize.css') }}" />
    <link rel="stylesheet" href="{{ asset('css/register.css') }}" />
</head>

<body>
    <header class="header">
        <h2 class="header__title">FashionablyLate</h2>
        <a href="/login" class="header__link">login</a>
    </header>
    <main>
        <div class="register-form">
            <h2 class="register-form__header">Register</h2>
            {{-- フォーム自体を囲むwrapper --}}
            <div class="register-form__inner">
                <form class="register-form__form" method="post" action="/register">
                    @csrf
                    {{-- お名前 --}}
                    <div class="register-form__group">
                        <label class="register-form__label" for="name">お名前</label>
                        <input class="register-form__input" type="text" id="name" name="name"
                            placeholder="テストたろう" value="{{ old('name') }}">
                        <p class="register-form__error">
                            @error('name')
                                {{ $message }}
                            @enderror
                        </p>
                    </div>
                    {{-- メールアドレス --}}
                    <div class="register-form__group">
                        <label class="register-form__label" for="email">メールアドレス</label>
                        <input class="register-form__input" type="email" id="email" name="email"
                            placeholder="test@gmail.com" value="{{ old('email') }}">
                        <p class="register-form__error">
                            @error('email')
                                {{ $message }}
                            @enderror
                        </p>
                    </div>
                    {{-- パスワード --}}
                    <div class="register-form__group">
                        <label class="register-form__label" for="password">パスワード</label>
                        <input class="register-form__input" type="password" id="password" name="password"
                            placeholder="test1234" value="{{ old('password') }}">
                        <p class="register-form__error">
                            @error('password')
                                {{ $message }}
                            @enderror
                        </p>
                    </div>
                    {{-- 登録ボタン --}}
                    <button class="register-form__button" type="submit">登録</button>
                </form>
            </div>
        </div>
    </main>
</body>

</html>
