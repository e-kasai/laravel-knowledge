### 説明
主モデルの定義の仕方テンプレート

### code

```php

<?php

// 名前空間 他の場所から使うときもこれを書くとフルパス不要
namespace App\Models;

// トレイト（機能の横断的再利用）を取り込む宣言
// `HasFactory` を `use` することでこのモデルでファクトリを使えるようになります。
use Illuminate\Database\Eloquent\Factories\HasFactory;

// Eloquentの基底クラス `Model` をインポート
use Illuminate\Database\Eloquent\Model;
//子モデルクラスをインポート
use App\Models\Contact;


// `Model` の機能を継承した具体的なデータオブジェクト`ex:Category`
class Category extends Model
{
	// このモデル用のファクトリを使う
    use HasFactory;

	// fillable設定
    protected $fillable = [
        'content'
    ];

	// リレーション設定
    public function items()
{
    return $this->hasMany(Item::class, 'seller_id'); 
    // 第2引数で外部キー指定。デフォルトは user_id だからカスタマイズ必須
    //　FKがtebale名_id出ない場合のみ第２引数にFKを書く
}
}
